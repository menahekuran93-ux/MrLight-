// Utility to escape HTML to prevent XSS function escapeHTML(str) { return String(str) .replace(/&/g, '&') .replace(//g, '>') .replace(/"/g, '"') .replace(/'/g, '''); } // Adds an agent card safely and caps feed length function addCard(evt) { const feed = document.getElementById('feed'); const c = PC[evt.position] || '#888'; const name = escapeHTML(evt.name.toUpperCase()); const argument = escapeHTML(evt.argument); const keySignal = escapeHTML(evt.key_signal); const round = evt.round || 1; const count = evt.count || 1; const confidence = Math.round((evt.confidence || 0) * 100); const div = document.createElement('div'); div.className = 'card'; div.style.borderLeft = `3px solid ${c}`; div.innerHTML = `
${name} ${PI[evt.position] || ''} ${evt.position.toUpperCase()} ${confidence}% ${evt.flipped ? '⚡ FLIPPED' : ''} R${round} · ${count} agents
${argument}
KEY SIGNAL: ${keySignal}
`; feed.appendChild(div); // Keep only the latest 100 cards to avoid performance issues while (feed.children.length > 100) feed.removeChild(feed.firstChild); feed.scrollTop = feed.scrollHeight; document.getElementById('feedTitle').textContent = `LIVE DEBATE FEED · ${feed.children.length} TRANSMISSIONS`; } // Updates faction bar with rounding correction and safe handling function updateFaction(tally, flips) { const total = (tally.bullish || 0) + (tally.bearish || 0) + (tally.neutral || 0); if (!total) return; // Calculate percentages with rounding correction let bull = Math.round((tally.bullish || 0) / total * 100); let bear = Math.round((tally.bearish || 0) / total * 100); let neut = 100 - bull - bear; // Ensure percentages sum to exactly 100 const sum = bull + bear + neut; if (sum !== 100) neut += 100 - sum; document.getElementById('factionBar').style.display = 'block'; document.getElementById('barBull').style.width = `${bull}%`; document.getElementById('barNeut').style.width = `${neut}%`; document.getElementById('barBear').style.width = `${bear}%`; document.getElementById('bullLabel').textContent = `▲ BULL ${bull}% (${tally.bullish || 0})`; document.getElementById('neutLabel').textContent = `◆ NEUTRAL ${neut}% (${tally.neutral || 0})`; document.getElementById('bearLabel').textContent = `▼ BEAR ${bear}% (${tally.bearish || 0})`; if (flips && flips.length) { document.getElementById('flipsMsg').textContent = `⚡ ${flips.length} group(s) flipped: ${flips.map(f => `${escapeHTML(f.name)} (${f.from}→${f.to})`).join(', ')}`; } }
